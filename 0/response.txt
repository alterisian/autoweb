# app/controllers/websites_controller.rb
class WebsitesController < ApplicationController
  def new
  end

  def create
    @name = params[:name]
    @description = params[:description]

    # Implement AI logic here to generate website content

    # Deploy to Vercel via API
    deploy_to_vercel(@name, @description)

    # Deploy to Fly.io
    deploy_to_flyio

    redirect_to root_path, notice: 'Website generated and deployed successfully!'
  end

  private

  def deploy_to_vercel(name, description)
    response = HTTParty.post('https://api.vercel.com/v2/now/deployments', {
      headers: { "Authorization" => "Bearer YOUR_VERCEL_API_TOKEN" },
      body: {
        name: name,
        description: description,
        files: [
          { file: "index.html", data: "<html><body><h1>#{name}</h1><p>#{description}</p></body></html>" }
        ]
      }.to_json
    })
    puts response.body
  end

  def deploy_to_flyio
    system("fly deploy")
  end
end

# fly.toml
app = "ai-website-generator"

[[services]]
  internal_port = 8080
  protocol = "tcp"

  [[services.ports]]
    handlers = ["http"]
    port = 80

  [[services.ports]]
    handlers = ["tls", "http"]
    port = 443

  [[services.tcp_checks]]
    grace_period = "1s"
    interval = "15s"
    restart_limit = 0
    timeout = "2s"
Deploy to Fly.io:

bash
Copy code
